# `std::ui` Redesign + REPL Widget Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Replace the imperative `std::ui` with a two-layer declarative API (low-level builders + opinionated `repl()` widget) over the existing `lib/tui/` engine, and rewrite the agency-agent's REPL loop onto the new widget as the acceptance test.

**Architecture:** A thin TS bridge (`lib/stdlib/ui.ts`) exposes the existing `lib/tui/` engine's `Screen.runLoop`, `nextKey`, and rendering primitives. Layer 1 (pure Agency) wraps those as builder functions (`column`, `row`, `box`, `line`, `text`, `list`, `textInput`) plus a `Builder` receiver type that lets users assemble element trees with `column() as col { col.row(...) as r { ... } }`. Layer 2 (pure Agency) builds an opinionated `repl()` widget on top of Layer 1 that bundles status line, scroll output, command palette, and input history. The agency-agent's REPL collapses onto `repl()`.

**Tech Stack:** Agency (the language), TypeScript (TS bridge + test harnesses), vitest, the existing `lib/tui/` declarative TUI engine.

**Spec:** [docs/superpowers/specs/2026-05-30-stdlib-ui-redesign-design.md](../specs/2026-05-30-stdlib-ui-redesign-design.md)

**Working directory:** `packages/agency-lang/` (all paths below are relative to it).

**Documentation invariant:** Per the spec and per `AGENTS.md`, every public type and function in `stdlib/ui.agency` MUST have a `/** ... */` doc comment, and the file MUST open with a `/** @module */` block including usage examples. The `make doc` step at the end of each milestone regenerates `docs/site/stdlib/ui.md` and verifies the docs render.

```ts
async runLoop<S>(opts: {
  initialState: S;
  render: (state: S) => Element;
  handleKey: (state: S, event: KeyEvent) => S;
  isDone: (state: S) => boolean;
  label?: string;
  tickMs?: number;
}): Promise<S> {
  let state = opts.initialState;
  this.render(opts.render(state), opts.label);
  while (!opts.isDone(state)) {
    if (opts.tickMs !== undefined) {
      // Race the next keypress against a tick timer. On tick, re-render
      // with the current state (no transition). On keypress, run handleKey
      // then re-render.
      const tickPromise = new Promise<"__tick__">((resolve) =>
        setTimeout(() => resolve("__tick__"), opts.tickMs),
      );
      const result = await Promise.race([
        this.nextKey().then((ev) => ({ kind: "key" as const, ev })),
        tickPromise.then(() => ({ kind: "tick" as const })),
      ]);
      if (result.kind === "key") {
        state = opts.handleKey(state, result.ev);
      }
    } else {
      const event = await this.nextKey();
      state = opts.handleKey(state, event);
    }
    this.render(opts.render(state), opts.label);
  }
  return state;
}
```

NB: the tick-side `nextKey()` Promise is *leaked* (still pending after the race resolves). The next loop iteration will await the same `nextKey()` promise — `ScriptedInput`'s waiter queue and `TerminalInput`'s underlying readline both tolerate multiple waiters, so this is safe. If a later cleanup pass wants to cancel the leaked promise, add an abort signal then.

```ts
import {
  Screen,
  TerminalInput,
  TerminalOutput,
  type Element,
  type KeyEvent,
  type InputSource,
  type OutputTarget,
} from "@/tui/index.js";

// Module-level handles. Tests inject scripted variants via the
// `_setInputSource` / `_setOutputTarget` / `_setSize` helpers below.
let inputSource: InputSource | null = null;
let outputTarget: OutputTarget | null = null;
let width = 80;
let height = 24;

// Default to terminal I/O at the actual TTY size unless tests have
// already injected substitutes.
function ensureScreen(): Screen {
  if (!inputSource) inputSource = new TerminalInput();
  if (!outputTarget) outputTarget = new TerminalOutput();
  if (process.stdout.isTTY) {
    width = process.stdout.columns || width;
    height = process.stdout.rows || height;
  }
  return new Screen({ input: inputSource, output: outputTarget, width, height });
}

export function _setInputSource(src: InputSource): void { inputSource = src; }
export function _setOutputTarget(out: OutputTarget): void { outputTarget = out; }
export function _setSize(w: number, h: number): void { width = w; height = h; }

// Active screen registered by `repl()` so the policy handler can probe.
let activeScreenRef: Screen | null = null;
export function _setActiveScreen(s: Screen | null): void { activeScreenRef = s; }
export function _hasActiveScreen(): boolean { return activeScreenRef !== null; }

export async function _runLoop(
  initialState: any,
  renderFn: (state: any) => any,
  handleKeyFn: (state: any, ev: KeyEvent) => any,
  isDoneFn: (state: any) => boolean,
  tickMs?: number,
): Promise<any> {
  const screen = ensureScreen();
  activeScreenRef = screen;
  try {
    return await screen.runLoop({
      initialState,
      render: (s) => renderFn(s) as Element,
      handleKey: (s, ev) => handleKeyFn(s, ev),
      isDone: (s) => isDoneFn(s),
      tickMs,
    });
  } finally {
    activeScreenRef = null;
    screen.destroy();
  }
}

export function _renderOnce(element: any): void {
  const screen = ensureScreen();
  screen.render(element as Element);
}

export async function _readKey(): Promise<KeyEvent> {
  const screen = activeScreenRef ?? ensureScreen();
  return screen.nextKey();
}

// Writes one text line to the active REPL's scroll buffer. Called by
// `_routePrompt` in Agency when there is an active screen. Implementation
// stub for now — wired through to the actual REPL state in Milestone D.
export function _appendToActiveScroll(text: string): void {
  // Real impl is registered by repl() via `_setScrollAppender` below.
  scrollAppender?.(text);
}

let scrollAppender: ((text: string) => void) | null = null;
export function _setScrollAppender(fn: ((text: string) => void) | null): void {
  scrollAppender = fn;
}

// `_promptFromChoices` reads keys until the user types an entry from
// `choices`, returns that entry. Lives here (not in Agency) because
// it has to bypass the REPL's normal input handling.
export async function _promptFromChoices(
  text: string,
  choices: string[],
): Promise<string> {
  _appendToActiveScroll(text);
  // Read line-shaped input from whichever source is wired up. The
  // ScriptedInput / TerminalInput both implement `nextLine`.
  const src = activeScreenRef ? (inputSource ?? new TerminalInput()) : new TerminalInput();
  for (;;) {
    const answer = (await src.nextLine("> ")).trim();
    if (choices.includes(answer)) return answer;
    _appendToActiveScroll(`(invalid; one of ${choices.join("/")})`);
  }
}
```

```ts
import process from "process";

// All ANSI escapes for the scroll-region mechanism live in this file.
// Everything else in lib/stdlib/ goes through the helpers below or
// renders via lib/tui/.

const ESC = "\x1b";
const CSI = `${ESC}[`;

let installedBottomRows = 0;
let scrollBottom = 0;
let resizeHandler: (() => void) | null = null;

/** Row (1-indexed) where the scroll region ends and the bottom region
 *  begins. Returns 0 before `installRegion` runs or in non-TTY mode. */
export function scrollBottomRow(): number {
  return scrollBottom;
}

/** Install the terminal scroll region so the top `H - bottomRows`
 *  rows scroll natively (preserving scrollback) and the bottom
 *  `bottomRows` rows are reserved for the TUI bottom region.
 *  No-op when stdout is not a TTY. */
export function installRegion(bottomRows: number): void {
  installedBottomRows = bottomRows;
  if (!process.stdout.isTTY) {
    scrollBottom = 0;
    return;
  }
  const rows = process.stdout.rows ?? 24;
  scrollBottom = Math.max(1, rows - bottomRows);
  process.stdout.write(`${CSI}1;${scrollBottom}r`);
}

/** Reset the terminal to its default scroll region and print a
 *  trailing newline so the next shell prompt lands below the bottom
 *  region. Idempotent. */
export function resetRegion(): void {
  if (!process.stdout.isTTY) {
    installedBottomRows = 0;
    scrollBottom = 0;
    return;
  }
  process.stdout.write(`${CSI}r`);
  process.stdout.write("\n");
  installedBottomRows = 0;
  scrollBottom = 0;
}

/** Save cursor, move to the start of the bottom region, run `fn`
 *  (which should write a frame to stdout), restore cursor. Used by
 *  `BottomRegionOutputTarget` to land TUI frames in the right spot
 *  without leaking cursor state into the scroll region. */
export function withBottomCursor(fn: () => void): void {
  if (!process.stdout.isTTY) {
    fn();
    return;
  }
  process.stdout.write(`${CSI}s`); // save cursor
  process.stdout.write(`${CSI}${scrollBottom + 1};1H`); // move to bottom region
  fn();
  process.stdout.write(`${CSI}u`); // restore cursor
}

/** Recompute the scroll region after a terminal resize. Call this
 *  from a SIGWINCH handler. */
export function onResize(): void {
  if (installedBottomRows > 0) {
    installRegion(installedBottomRows);
  }
}

/** Install a SIGWINCH listener that re-installs the region. Returns
 *  a teardown function. Tests may invoke `onResize()` directly
 *  instead. */
export function installResizeHandler(): () => void {
  if (resizeHandler) return () => {};
  resizeHandler = () => onResize();
  process.stdout.on("resize", resizeHandler);
  return () => {
    if (resizeHandler) {
      process.stdout.removeListener("resize", resizeHandler);
      resizeHandler = null;
    }
  };
}
```

```ts
import { withBottomCursor } from "./ui-region.js";
import type { OutputTarget, Frame } from "@/tui/index.js";

/**
 * Wraps any `OutputTarget` so frame writes land at the bottom of the
 * real terminal — used by `repl()`'s hybrid rendering. The wrapped
 * inner target sees a normal write call; `withBottomCursor`
 * positions the cursor before and restores it after.
 */
export class BottomRegionOutputTarget implements OutputTarget {
  constructor(private inner: OutputTarget) {}
  write(frame: Frame, label?: string): void {
    withBottomCursor(() => {
      this.inner.write(frame, label);
    });
  }
  destroy(): void {
    if (this.inner.destroy) this.inner.destroy();
  }
}

/**
 * Append a raw text line to the scroll region. Used by `repl()` to
 * stream output lines into the terminal's native scrollback. Plain
 * `process.stdout.write` — no ANSI; the scroll region installed by
 * `ui-region` does the rest.
 */
export function _writeScrollLine(text: string): void {
  process.stdout.write(text + "\n");
}
```

  ```ts
  type LogState = { logs: string[]; cursor: number; done: boolean }

  def view(s: LogState): Element {
  ```

  ## How builders work

  Container builders (`column`, `row`, `box`) take a trailing block
  that receives a fresh `Builder`. Methods on the Builder append
  children to the parent in source order. Container methods on a
  Builder (`col.row`, `col.column`, `col.box`) take their own
  trailing block; leaf methods (`col.line`, `col.text`, `col.list`,
  `col.textInput`) don't.

  builder — there is no nested `Style` record to assemble.

```agency
 * @param tickMs       - Milliseconds between forced re-renders (omit for event-driven only)
 */
export def runLoop(
  initialState: any,
  render: (any) -> Element,
  handleKey: (any, KeyEvent) -> any,
  isDone: (any) -> boolean,
  tickMs?: number,
): any {
  return _runLoop(initialState, render, handleKey, isDone, tickMs)
```

```agency
// ---- Layer 2: repl() widget ----

import { read, write, exists } from "std::shell"
import { _installScrollRegion, _resetScrollRegion, _writeScrollLine,
         _runLoopHybrid } from "agency-lang/stdlib-lib/ui.js"

// Module-level: an active REPL's scroll appender. Other stdlib
// modules (notably std::policy) check this via `_hasActiveScreen()`
// and call `_routePrompt` to land their prompts in the right place.
// See the spec for the cross-module contract.
let _replOutputLines: string[] = []

type _ReplState = {
  buffer: string;
  history: string[];
  historyIdx: number;        // == history.length means "fresh entry"
  paletteOpen: boolean;
  paletteFilter: string;
  paletteCursor: number;
  done: boolean;
  // Hybrid-mode bookkeeping: tracks how many lines from `output()`
  // we've already streamed to the terminal's scroll region so each
  // tick only writes the new tail.
  outputLinesWritten: number;
  // Config (closed over so view + reduce can see them)
  prompt: string;
  paletteCommands: Record<string, string>;
  historyFile: string;
  historyMax: number;
  output: () -> string[];
  status: () -> { left: string, right: string };
  onSubmit: (string) -> boolean;
}

def _filteredPaletteKeys(s: _ReplState): string[] {
  let out: string[] = []
  for (k in keys(s.paletteCommands)) {
    if (s.paletteFilter == "" || k.toLowerCase().includes(s.paletteFilter.toLowerCase())) {
      out.push(k)
    }
  }
  return out
}

// In hybrid mode the scroll lines are owned by the terminal's native
// scroll region — they are written via `_writeScrollLine` from
// `_appendNewScrollLines`, not rendered into this tree. The tree only
// describes the bounded bottom region (palette + status + input).
def _replView(s: _ReplState): Element {
  const stat = s.status()
  const palKeys = _filteredPaletteKeys(s)
  const palItems: string[] = []
  for (k in palKeys) {
    palItems.push("${k}  — ${s.paletteCommands[k]}")
  }
  return column() as col {
    col.list(items: palItems, selectedIndex: s.paletteCursor,
             visible: s.paletteOpen, height: 5, border: true)
    col.row(bg: "blue", fg: "white", height: 1) as st {
      st.line(stat.left)
      st.box(flex: 1)
      st.line(stat.right)
    }
    col.row(height: 1) as ir {
      ir.line(s.prompt)
      ir.textInput(value: s.buffer, flex: 1)
    }
  }
}

// Per-tick: append any new tail of `output()` to the scroll region.
// Returns the updated state with `outputLinesWritten` advanced. If
// `output()` shrinks (e.g. user clears the conversation), we reset
// the counter and write the whole list afresh — the spec calls for
// a screen clear here too, but stdout-only append keeps the impl
// straightforward at v1; `/clear` consumers can clear manually.
def _appendNewScrollLines(s: _ReplState): _ReplState {
  const lines = s.output()
  if (lines.length < s.outputLinesWritten) {
    // Output shrunk; rewrite from scratch.
    for (ln in lines) { _writeScrollLine(ln) }
    return { ...s, outputLinesWritten: lines.length }
  }
  if (lines.length > s.outputLinesWritten) {
    for (i in range(s.outputLinesWritten, lines.length)) {
      _writeScrollLine(lines[i])
    }
    return { ...s, outputLinesWritten: lines.length }
  }
  return s
}

def _persistHistory(file: string, history: string[], max: number) {
  if (file == "") { return }
  let toSave = history
  if (history.length > max) {
    toSave = history.slice(history.length - max, history.length)
  }
  try write("", file, toSave.join("\n")) with approve
}

def _replReduce(s: _ReplState, k: KeyEvent): _ReplState {
  // Palette controls take priority when the palette is open.
  if (s.paletteOpen) {
    const palKeys = _filteredPaletteKeys(s)
    if (k.key == "escape") { return { ...s, paletteOpen: false, paletteFilter: "" } }
    if (k.key == "up")     { return { ...s, paletteCursor: max(0, s.paletteCursor - 1) } }
    if (k.key == "down")   { return { ...s, paletteCursor: min(palKeys.length - 1, s.paletteCursor + 1) } }
    if (k.key == "enter" || k.key == "tab") {
      if (palKeys.length == 0) { return { ...s, paletteOpen: false, paletteFilter: "" } }
      return { ...s, buffer: palKeys[s.paletteCursor], paletteOpen: false, paletteFilter: "" }
    }
    if (k.key == "backspace") {
      if (s.paletteFilter == "") { return { ...s, paletteOpen: false } }
      return { ...s, paletteFilter: s.paletteFilter.slice(0, s.paletteFilter.length - 1), paletteCursor: 0 }
    }
    // Printable: append to filter.
    if (k.key.length == 1) {
      return { ...s, paletteFilter: s.paletteFilter + k.key, paletteCursor: 0 }
    }
    return s
  }

  // Palette closed: normal editing.
  if (k.ctrl && k.key == "c") { return { ...s, done: true } }
  if (k.key == "/" && s.buffer == "") {
    return { ...s, paletteOpen: true, paletteFilter: "", paletteCursor: 0 }
  }
  if (k.key == "up" && s.historyIdx > 0) {
    return { ...s, historyIdx: s.historyIdx - 1, buffer: s.history[s.historyIdx - 1] }
  }
  if (k.key == "down" && s.historyIdx < s.history.length) {
    const nxt = s.historyIdx + 1
    const buf = nxt < s.history.length ? s.history[nxt] : ""
    return { ...s, historyIdx: nxt, buffer: buf }
  }
  if (k.key == "backspace") {
    if (s.buffer == "") { return s }
    return { ...s, buffer: s.buffer.slice(0, s.buffer.length - 1) }
  }
  if (k.key == "enter") {
    const line = s.buffer
    const nextHist = [...s.history, line]
    _persistHistory(s.historyFile, nextHist, s.historyMax)
    const keepGoing = s.onSubmit(line)
    // After onSubmit runs the consumer typically pushed new lines into
    // its `output()` source. Drain the new tail to the scroll region
    // immediately so the user sees the response without waiting for
    // the next tick.
    const drained = _appendNewScrollLines({ ...s, buffer: "", history: nextHist,
                                            historyIdx: nextHist.length,
                                            done: !keepGoing })
    return drained
  }
  if (k.key.length == 1) {
    return { ...s, buffer: s.buffer + k.key }
  }
  return s
}

def _replIsDone(s: _ReplState): boolean { return s.done }

def _loadHistory(file: string): string[] {
  if (file == "") { return [] }
  if (!exists("", file)) { return [] }
  const r = try read("", file) with approve
  if (isFailure(r)) { return [] }
  return r.value.split("\n")
}

/**
 * Drop-in REPL widget for interactive CLI agents. Bundles a scrollable
 * output area, a status line, a command palette (triggered by `/`),
 * and an input line with persistent history.
 *
 * Lifecycle: enters raw mode, runs `runLoop` with `tickMs` so the
 * status updates while a turn is in flight, persists history on each
 * submit, restores the terminal on exit (including on exceptions).
 *
 * @param prompt - String shown before the input buffer
 * @param output - Re-evaluated every render to populate the scroll area
 * @param status - Re-evaluated every render to populate the status line
 * @param historyFile - `""` for in-memory only; otherwise persist newline-delimited
 * @param historyMax - Trim oldest entries beyond this count on persist
 * @param paletteCommands - Map of `/cmd` → description; iterated in insertion order
 * @param onSubmit - Called with the submitted line; return `false` to exit the loop
 * @param tickMs - Render cadence in milliseconds (default 100)
 */
// Bottom-region row count for hybrid mode. v1: status (1) + input (1).
// When the palette is open we add `_PALETTE_ROWS` and reissue
// `_installScrollRegion` from within `_replReduce`; reverted on close.
const _BASE_BOTTOM_ROWS: number = 2
const _PALETTE_ROWS: number = 5

def _bottomRows(s: _ReplState): number {
  return s.paletteOpen ? _BASE_BOTTOM_ROWS + _PALETTE_ROWS : _BASE_BOTTOM_ROWS
}

export def repl(
  prompt: string = "> ",
  output: () -> string[],
  status: () -> { left: string, right: string },
  historyFile: string = "",
  historyMax: number = 1000,
  paletteCommands: Record<string, string> = {},
  onSubmit: (string) -> boolean,
  tickMs: number = 100,
) {
  const initial: _ReplState = {
    buffer: "",
    history: _loadHistory(historyFile),
    historyIdx: 0,
    paletteOpen: false,
    paletteFilter: "",
    paletteCursor: 0,
    done: false,
    outputLinesWritten: 0,
    prompt: prompt,
    paletteCommands: paletteCommands,
    historyFile: historyFile,
    historyMax: historyMax,
    output: output,
    status: status,
    onSubmit: onSubmit,
  }
  initial.historyIdx = initial.history.length

  // Install the terminal scroll region for hybrid rendering. The
  // top region keeps native scrollback; the bottom `_BASE_BOTTOM_ROWS`
  // rows are owned by the bounded `Screen` driven by `_runLoopHybrid`.
  _installScrollRegion(_BASE_BOTTOM_ROWS)
  // Seed: stream any pre-existing output() lines to the scroll
  // region so a non-empty initial transcript shows immediately.
  const seeded = _appendNewScrollLines(initial)

  // `_runLoopHybrid` wraps the active `OutputTarget` in
  // `BottomRegionOutputTarget` and bounds the `Screen` to the bottom
  // region. `tickMs` drives the live status updates. The handler
  // wrapping ensures `_resetScrollRegion()` runs on every exit path
  // (clean exit, onSubmit -> false, Ctrl-C, exception).
  handle _: AgencyError {
    _resetScrollRegion()
    raise _
  }
  try {
    _runLoopHybrid(seeded,
                   render: _replView,
                   handleKey: _replReduce,
                   isDone: _replIsDone,
                   tickMs: tickMs,
                   bottomRows: _BASE_BOTTOM_ROWS)
  }
  _resetScrollRegion()
}
```

```agency
/**
 * Private — used by other stdlib modules (notably `std::policy`)
 * to route a prompt through the active REPL when one exists, or
 * fall back to raw `print` + `input` otherwise.
 *
 * Not part of the public API; the underscore prefix is convention,
 * but the export is required so cross-module imports work.
 */
export def _routePrompt(text: string, choices: string[]): string {
  if (_hasActiveScreen()) {
    return _promptFromChoices(text, choices)
  }
  print(text)
  let answer = ""
  let valid = false
  while (!valid) {
    answer = input("> ").trim()
    if (choices.includes(answer)) { valid = true }
    else { print("(one of ${choices.join(\"/\")})") }
  }
  return answer
}
```
